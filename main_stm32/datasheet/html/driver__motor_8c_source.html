44static inline uint16_t motor_map_percent(Motor_Handle_t *hmotor, int16_t percent){
45 if (percent < 0) percent = 0;
46 if (percent > 100) percent = 100;
47
48 const uint16_t min = hmotor->min_pulse_ticks;
49 const uint16_t max = hmotor->max_pulse_ticks;
50
51 return (uint16_t)(min + ((uint32_t)(max - min) * (uint32_t)percent) / 100u);

61static uint8_t motor_speed_mms_to_pwm_percent(Motor_Handle_t *hmotor, int16_t value){
62 if (value >= hmotor->max_speed_pos_mms) return 100;
63 if (value <= hmotor->max_speed_neg_mms) return 0;
64
65 if (value >= 0)
66 return (uint8_t)(50 + ((int32_t)value * 50) / hmotor->max_speed_pos_mms);
67 else

140void motor_process_1ms(Motor_Handle_t *hmotor, uint32_t now_ms){
141 if (!hmotor) return;
142
143 const bool want_neutral = (hmotor->ctx.target_speed_mms == 0);
144
145 switch (hmotor->state){
146 case MOTOR_STATE_NEUTRAL:
147 motor_pwm_percent(hmotor, PWM_NEUTRAL);
148
149 if(!want_neutral){
150 if(hmotor->ctx.target_forward){
151 motor_pwm_percent(hmotor, hmotor->ctx.target_pwm);
152 hmotor->go_forward = true;
153 hmotor->state = MOTOR_STATE_FORWARD_HOLD;
154 }
155 else{
156 motor_pwm_percent(hmotor, PWM_BRAKE_REV);
157 hmotor->ctx.deadline_ms = now_ms + T_BRAKE_MS;
158 hmotor->state = MOTOR_STATE_NEUTRAL_TO_REVERSE_TAP;
159 }
160 }
161 break;
162
163 case MOTOR_STATE_NEUTRAL_TO_REVERSE_TAP:
164 if (!time_reached(now_ms, hmotor->ctx.deadline_ms)) break;
165 motor_pwm_percent(hmotor, PWM_NEUTRAL);
166 hmotor->ctx.deadline_ms = now_ms + T_NEUTRAL_GAP_MS;
167 hmotor->state = MOTOR_STATE_NEUTRAL_TO_REVERSE_GAP;
168 break;
169
170 case MOTOR_STATE_NEUTRAL_TO_REVERSE_GAP:
171 if(!time_reached(now_ms, hmotor->ctx.deadline_ms)) break;
172 if(want_neutral){
173 motor_pwm_percent(hmotor, PWM_NEUTRAL);
174 hmotor->state = MOTOR_STATE_NEUTRAL;
175 }
176 else if (hmotor->ctx.target_forward){

178 hmotor->go_forward = true;
