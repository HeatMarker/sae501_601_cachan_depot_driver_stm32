117static void parse_byte(uint8_t b){
118 switch(st){
119 case S_HDR:hdr=b;st=S_D0;break;
120 case S_D0:d0=b;st=S_D1;break;
121 case S_D1:d1=b;st=S_CRC;break;
122 case S_CRC:{
123 uint8_t buf[3]={hdr,d0,d1};
124 uint8_t crc=serial_crc8_atm(buf,3);
125 if(crc==b)handle_frame(hdr,d0,d1);
126 st=S_HDR;
127 }
128 break;
129 default:st=S_HDR;break;
130 }
