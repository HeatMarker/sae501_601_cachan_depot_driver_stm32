101static int8_t bmi088_spi_write(uint8_t reg_addr, const uint8_t *reg_data, uint32_t len, void *intf_ptr){
102 bmi088_cs_t *cs = (bmi088_cs_t*)intf_ptr;
103
104 HAL_GPIO_WritePin(cs->port, cs->pin, GPIO_PIN_RESET);
105
106 uint8_t addr = reg_addr & 0x7F;
107 HAL_SPI_Transmit(bmi088_hspi, &addr, 1, HAL_MAX_DELAY);
108
