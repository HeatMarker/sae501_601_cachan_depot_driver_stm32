99static uint32_t GetMicrosTotal(void){
100 uint32_t m_overflow;
101 uint16_t m_counter;
102
103 do{
104 m_overflow = tim3_overflow_cnt;
105 m_counter = LL_TIM_GetCounter(TIM3);
106 }
107 while(m_overflow != tim3_overflow_cnt);
108
109 return (m_overflow << 16) + m_counter;
110}
