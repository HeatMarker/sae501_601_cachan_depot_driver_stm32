52static inline uint16_t servo_map_percent(Servo_Handle_t *hservo, int16_t percent){
53 int16_t corrected = percent + SERVO_OFFSET_PERCENT;
54
55 if (corrected < 0) corrected = 0;
56 if (corrected > 100) corrected = 100;
57
58 uint16_t min = hservo->min_pulse_ticks;
59 uint16_t max = hservo->max_pulse_ticks;
60 return min + ((max - min) * corrected) / 100;
61}

97void servo_pwm_angle_abs_value(Servo_Handle_t *hservo, uint16_t abs_value){
98 int32_t angle_centi = map(abs_value, 0, 65535, -4500, 4500);
99
100 if (angle_centi < -2000) angle_centi = -2000;
