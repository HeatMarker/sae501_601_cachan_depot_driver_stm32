134int serial_write_nb(const uint8_t *data,uint16_t len){
135 uint16_t written=0;
136 while(written<len){
137 uint32_t space=tx_space();
138 if(space==0)return(written>0)?(int)written:-EWOULDBLOCK;
139 uint32_t head=tx_head;
140 uint32_t room_linear=(head>=tx_tail)?(TX_RING_SIZE-head-((tx_tail==0)?1u:0u)):(tx_tail-head-1u);
141 if(room_linear==0)return(written>0)?(int)written:-EWOULDBLOCK;
142 uint32_t to_copy=len-written;
143 if(to_copy>room_linear)to_copy=room_linear;
144 if(to_copy>space)to_copy=space;
145 memcpy(&tx_ring[head],&data[written],to_copy);
146 tx_head=(head+to_copy)&TX_RING_MASK;
147 written+=(uint16_t)to_copy;
148 }
149 serial_kick_tx();
150 return(int)written;
151}

250size_t serial_read_until(uint8_t *dst,size_t max_len,uint8_t delim){
251 if(!max_len)return 0;
252 uint32_t head=rx_head,tail=rx_tail;
253 if(head==tail)return 0;
254 uint32_t i=tail;
255 while(i!=head){
256 if(rx_ring[i]==delim){
257 size_t msg_len=(i>=tail)?(i-tail+1u):(SERIAL_RX_RING_SIZE-tail+i+1u);
258 if(msg_len>max_len)return 0;

263 rx_tail=(tail+msg_len)&RING_MASK;
264 return msg_len;
265 }
266 i=(i+1u)&RING_MASK;
